
 Tabella Dipartimento (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descrizione TEXT
);

 Tabella Corso_di_Laurea (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descrizione TEXT,
    id_dipartimento INT NOT NULL,
    FOREIGN KEY (id_dipartimento) REFERENCES Dipartimento(id)
);

 Tabella Corso (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descrizione TEXT,
    cfu INT,
    anno INT,
    id_corso_di_laurea INT NOT NULL,
    FOREIGN KEY (id_corso_di_laurea) REFERENCES Corso_di_Laurea(id)
);

 Tabella Insegnante (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cognome VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE
);

 Tabella Corso_Insegnante (
    id_corso INT,
    id_insegnante INT,
    PRIMARY KEY (id_corso, id_insegnante),
    FOREIGN KEY (id_corso) REFERENCES Corso(id),
    FOREIGN KEY (id_insegnante) REFERENCES Insegnante(id)
);

 Tabella Studente (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cognome VARCHAR(100) NOT NULL,
    matricola VARCHAR(20) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE,
    data_iscrizione DATE,
    id_corso_di_laurea INT NOT NULL,
    FOREIGN KEY (id_corso_di_laurea) REFERENCES Corso_di_Laurea(id)
);

 Tabella Appello (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_corso INT NOT NULL,
    data_appello DATE NOT NULL,
    aula VARCHAR(50),
    FOREIGN KEY (id_corso) REFERENCES Corso(id)
);

CREATE TABLE Esame (
    id_studente INT,
    id_appello INT,
    voto TINYINT,
    lode BOOLEAN DEFAULT FALSE,
    PRIMARY KEY (id_studente, id_appello),
    FOREIGN KEY (id_studente) REFERENCES Studente(id),
    FOREIGN KEY (id_appello) REFERENCES Appello(id)
);
